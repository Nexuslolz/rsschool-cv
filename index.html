Convert string to camel case

                
                  function  toCamelCase(str){
                  let word 
                  for(i= 0;i<str.length;i++){
                  if(str[i]==='-'||str[i]==='_'){
                   word = str[i+1]
                   str = str.replace(word,word.toUpperCase())
                  }
                }
                str = str.replace(/[^a-zа-яё0-9\s]/gi,' ');
                str = str.replace(/\s/g, '');
                  return str
                }
